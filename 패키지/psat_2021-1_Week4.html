def getText(text) :  #    
    dummy_val = ['1','2','3','4','5','6','7','8','9','0','-','(',')'] #           
    result = ''
    for letter in text : 
        if letter in dummy_val : #    
            continue
        else : 
            result += letter #   
    return result
